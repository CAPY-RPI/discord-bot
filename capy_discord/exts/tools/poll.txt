
import logging

import discord
from discord import app_commands
from discord.ext import commands

from capy_discord.ui.embeds import info_embed


class Poll(commands.Cog):
    """Cog for voting and polls."""

    def __init__(self, bot: commands.Bot) -> None:
        """Initializing"""
        self.bot = bot
        self.log = logging.getLogger(__name__)

    @app_commands.command(name="poll", description="Create a poll with reaction emojis")
    @app_commands.describe(
        question="The question to ask",
        options="Optional: Comma-separated list of options (e.g. Yes, No, Maybe)",
    )
    async def poll(self, interaction: discord.Interaction, question: str, options: str | None = None) -> None:
        """Create a poll with reactions."""
        try:
            # Emoji list for multi-option polls
            emoji_list = ["1️⃣", "2️⃣", "3️⃣", "4️⃣", "5️⃣", "6️⃣", "7️⃣", "8️⃣", "9️⃣", "🔟"]

            if options:
                # Multi-option poll
                option_list = [opt.strip() for opt in options.split(",") if opt.strip()]

                if len(option_list) < 2:
                    await interaction.response.send_message(
                        "Please provide at least 2 options for a multi-choice poll.",
                        ephemeral=True,
                    )
                    return

                if len(option_list) > 10:
                    await interaction.response.send_message(
                        "You can only have up to 10 options.",
                        ephemeral=True,
                    )
                    return

                description = ""
                for idx, opt in enumerate(option_list):
                    description += f"{emoji_list[idx]} {opt}\n"

                embed = info_embed(title=question, description=description)
                embed.set_footer(text=f"Poll created by {interaction.user.display_name}")

                await interaction.response.send_message(embed=embed)
                message = await interaction.original_response()

                for i in range(len(option_list)):
                    await message.add_reaction(emoji_list[i])

            else:
                # Simple Yes/No/Maybe poll
                embed = info_embed(title=question, description="👍 Yes\n👎 No\n🤷 Maybe")
                embed.set_footer(text=f"Poll created by {interaction.user.display_name}")

                await interaction.response.send_message(embed=embed)
                message = await interaction.original_response()

                for emoji in ["👍", "👎", "🤷"]:
                    await message.add_reaction(emoji)

            self.log.info("/poll created by %s in %s", interaction.user.id, interaction.guild_id)

        except Exception:
            self.log.exception("/poll failed")
            if not interaction.response.is_done():
                await interaction.response.send_message(
                    "Failed to create poll. Make sure I have permission to add reactions.",
                    ephemeral=True,
                )
            else:
                await interaction.followup.send(
                    "Failed to add reactions to the poll. Check my permissions.",
                    ephemeral=True,
                )


async def setup(bot: commands.Bot) -> None:
    """Set up the Poll cog."""
    await bot.add_cog(Poll(bot))
